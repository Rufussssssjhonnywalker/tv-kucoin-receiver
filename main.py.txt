from __future__ import annotations

import base64
import hashlib
import hmac
import json
import os
import time
import uuid
from typing import Any

import requests
from fastapi import FastAPI, Request
from fastapi.responses import JSONResponse

app = FastAPI(title="TradingView KuCoin Receiver")


def env_str(name: str, default: str = "") -> str:
    return os.getenv(name, default).strip()


def env_bool(name: str, default: bool = False) -> bool:
    value = os.getenv(name)
    if value is None:
        return default
    return value.strip().lower() in {"1", "true", "yes", "on"}


WEBHOOK_SECRET = env_str("WEBHOOK_SECRET", "CHANGE_ME")

KUCOIN_API_KEY = env_str("KUCOIN_API_KEY")
KUCOIN_API_SECRET = env_str("KUCOIN_API_SECRET")
KUCOIN_API_PASSPHRASE = env_str("KUCOIN_API_PASSPHRASE")

KUCOIN_USE_TEST = env_bool("KUCOIN_USE_TEST", True)

# KuCoin Futures endpoints
KUCOIN_BASE_TEST = "https://api-sandbox-futures.kucoin.com"
KUCOIN_BASE_LIVE = "https://api-futures.kucoin.com"
KUCOIN_BASE_URL = KUCOIN_BASE_TEST if KUCOIN_USE_TEST else KUCOIN_BASE_LIVE

DEFAULT_LEVERAGE = env_str("DEFAULT_LEVERAGE", "1")
TRADING_ENABLED = env_bool("TRADING_ENABLED", True)

SESSION = requests.Session()
SESSION.headers.update({"Content-Type": "application/json"})


def now_ms() -> str:
    return str(int(time.time() * 1000))


def json_dumps(data: Any) -> str:
    return json.dumps(data, separators=(",", ":"), ensure_ascii=False)


def kucoin_headers(endpoint: str, method: str, body: str) -> dict[str, str]:
    ts = now_ms()
    prehash = f"{ts}{method.upper()}{endpoint}{body}"
    signature = base64.b64encode(
        hmac.new(
            KUCOIN_API_SECRET.encode("utf-8"),
            prehash.encode("utf-8"),
            hashlib.sha256,
        ).digest()
    ).decode("utf-8")

    passphrase = base64.b64encode(
        hmac.new(
            KUCOIN_API_SECRET.encode("utf-8"),
            KUCOIN_API_PASSPHRASE.encode("utf-8"),
            hashlib.sha256,
        ).digest()
    ).decode("utf-8")

    return {
        "KC-API-KEY": KUCOIN_API_KEY,
        "KC-API-SIGN": signature,
        "KC-API-TIMESTAMP": ts,
        "KC-API-PASSPHRASE": passphrase,
        "KC-API-KEY-VERSION": "2",
        "Content-Type": "application/json",
    }


def post_kucoin(endpoint: str, payload: dict[str, Any]) -> tuple[int, Any]:
    body = json_dumps(payload)
    headers = kucoin_headers(endpoint, "POST", body)
    url = f"{KUCOIN_BASE_URL}{endpoint}"
    response = SESSION.post(url, headers=headers, data=body, timeout=30)

    try:
        data = response.json()
    except Exception:
        data = {"raw_text": response.text}

    return response.status_code, data


def build_entry_order(signal: dict[str, Any]) -> dict[str, Any]:
    action = str(signal.get("action", "")).lower()
    symbol = str(signal.get("symbol", "")).strip()
    notional_usdt = float(signal.get("notional_usdt", 0))
    reduce_only = bool(signal.get("reduce_only", False))

    if action not in {"buy", "sell"}:
        raise ValueError(f"Unsupported entry action: {action}")

    if not symbol:
        raise ValueError("Missing symbol")

    if notional_usdt <= 0:
        raise ValueError("notional_usdt must be > 0")

    side = "buy" if action == "buy" else "sell"

    # Nota:
    # size/qty esatti dipendono dalle specifiche del contract.
    # Qui usiamo "funds" come notional-style payload semplice per il test.
    # Se il tuo vecchio main.py usava un altro campo, poi lo adattiamo.
    order = {
        "clientOid": signal.get("signal_id") or str(uuid.uuid4()),
        "side": side,
        "symbol": symbol,
        "type": "market",
        "leverage": str(signal.get("leverage", DEFAULT_LEVERAGE)),
        "reduceOnly": reduce_only,
        "remark": str(signal.get("strategy", "tv-webhook"))[:100],
        "funds": str(notional_usdt),
    }
    return order


def build_exit_order(signal: dict[str, Any]) -> dict[str, Any]:
    action = str(signal.get("action", "")).lower()
    symbol = str(signal.get("symbol", "")).strip()

    if action not in {"buy", "sell"}:
        raise ValueError(f"Unsupported exit action: {action}")

    if not symbol:
        raise ValueError("Missing symbol")

    # Per uscita market reduce-only.
    # Nota: senza size precisa, su alcuni setup KuCoin potrebbe richiedere un adattamento.
    order = {
        "clientOid": signal.get("signal_id") or str(uuid.uuid4()),
        "side": "buy" if action == "buy" else "sell",
        "symbol": symbol,
        "type": "market",
        "reduceOnly": True,
        "closeOrder": True,
        "remark": str(signal.get("strategy", "tv-webhook-exit"))[:100],
    }
    return order


@app.get("/")
def root() -> dict[str, Any]:
    return {
        "status": "ok",
        "service": "TradingView KuCoin Receiver",
        "mode": "test" if KUCOIN_USE_TEST else "live",
        "trading_enabled": TRADING_ENABLED,
    }


@app.get("/health")
def health() -> dict[str, Any]:
    return {"ok": True}


@app.post("/tv-signal")
async def tv_signal(request: Request) -> JSONResponse:
    try:
        payload = await request.json()
    except Exception:
        raw = await request.body()
        return JSONResponse(
            status_code=400,
            content={"status": "error", "message": "Invalid JSON", "raw": raw.decode("utf-8", errors="ignore")},
        )

    print("\n=== RECEIVED /tv-signal ===")
    print(json_dumps(payload))

    secret = str(payload.get("secret", "")).strip()
    if not WEBHOOK_SECRET:
        return JSONResponse(
            status_code=500,
            content={"status": "error", "message": "WEBHOOK_SECRET not configured"},
        )

    if secret != WEBHOOK_SECRET:
        return JSONResponse(
            status_code=401,
            content={"status": "error", "message": "Invalid secret"},
        )

    signal_type = str(payload.get("signal_type", "entry")).lower()

    if not TRADING_ENABLED:
        return JSONResponse(
            status_code=200,
            content={
                "status": "received_not_sent",
                "reason": "TRADING_ENABLED=false",
                "mode": "test" if KUCOIN_USE_TEST else "live",
                "signal_type": signal_type,
                "signal_id": payload.get("signal_id"),
            },
        )

    try:
        if signal_type == "entry":
            kucoin_payload = build_entry_order(payload)
        elif signal_type == "exit":
            kucoin_payload = build_exit_order(payload)
        else:
            raise ValueError(f"Unsupported signal_type: {signal_type}")
    except Exception as e:
        return JSONResponse(
            status_code=400,
            content={"status": "error", "message": str(e), "payload": payload},
        )

    print("=== FORWARD TO KUCOIN ===")
    print(json_dumps(kucoin_payload))

    endpoint = "/api/v1/orders"

    try:
        http_status, kucoin_response = post_kucoin(endpoint, kucoin_payload)
    except Exception as e:
        return JSONResponse(
            status_code=500,
            content={"status": "error", "message": f"KuCoin request failed: {e}"},
        )

    print(f"=== KUCOIN HTTP STATUS: {http_status} ===")
    print(json_dumps(kucoin_response))

    return JSONResponse(
        status_code=200,
        content={
            "status": "received_and_forwarded",
            "mode": "test" if KUCOIN_USE_TEST else "live",
            "signal_type": signal_type,
            "signal_id": payload.get("signal_id"),
            "kucoin_http_status": http_status,
            "kucoin_response": kucoin_response,
        },
    )
